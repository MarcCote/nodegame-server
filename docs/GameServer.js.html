GameServer.prototype.secureParse = function(msg) {

  try {
    var gameMsg = GameMsg.clone(JSON.parse(msg));
    this.log.msg('R, ' + gameMsg);
    return gameMsg;
  } catch (e) {
    this.log.log("Malformed msg received: " + e, 'ERR');
    return false;
  }

};
